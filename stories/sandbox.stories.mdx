import { Meta } from '@storybook/addon-docs/blocks';

import { Example } from '../';

export const defaultExample = `
render(() => {
    const [counter, setCounter] = React.useState(0);
    return (
        <div>
            <h2>Primary counter: {counter}</h2>
            <button type='button' onClick={() => setCounter((c) => c + 1)}>
                Increment
            </button>
        </div>
    );
})
`;

export const getCode = () => {
    const code = window.parent.location.search.split('/code=')[1];
    return code ? decodeURIComponent(code) : defaultExample;
};

<Meta
    title='Sandbox'
    parameters={{
        previewTabs: { canvas: { hidden: true } }
    }}
/>

<h1>Sandbox</h1>

Modify code and share it

<Example expanded={true} live={true} code={getCode()} />
